1819  DomainInsideSignature getDomainInside() const{
1820  auto rule = getRule();
1821  if ((rule == TasGrid::rule_gausshermite) || (rule == TasGrid::rule_gausshermiteodd)){ // unbounded domain
1822  return [](std::vector<double> const &)->bool{ return true; };
1823  }else if ((rule == TasGrid::rule_gausslaguerre) || (rule == TasGrid::rule_gausslaguerreodd)){ // bounded from below
1824  if (domain_transform_a.empty()){
1825  return [](std::vector<double> const &x)->bool{
1826  for(auto const &v : x) if (v < 0.0) return false;
1827  return true;
1828  };
1829  }else{
1830  size_t dims = (size_t) getNumDimensions();
1831  return [=](std::vector<double> const &x)->bool{
1832  for(size_t i=0; i<dims; i++) if (x[i] < domain_transform_a[i]) return false;
1833  return true;
1834  };
1835  }
1836  }else{
1837  if (domain_transform_a.empty()){
1838  if (isFourier()){
1839  return [](std::vector<double> const &x)->bool{
1840  for(auto const &v : x) if ((v < 0.0) || (v > 1.0)) return false;
1841  return true;
1842  };
1843  }else{
1844  return [](std::vector<double> const &x)->bool{
1845  for(auto const &v : x) if ((v < -1.0) || (v > 1.0)) return false;
1846  return true;
1847  };
1848  }
1849  }else{
1850  size_t dims = (size_t) getNumDimensions();
1851  return [=](std::vector<double> const &x)->bool{
1852  for(size_t i=0; i<dims; i++)
1853  if ((x[i] < domain_transform_a[i]) || (x[i] > domain_transform_b[i])) return false;
1854  return true;
