192  template<RuleLocal::erule eff_rule>
193  double evalBasisSupported(const int point[], const double x[], bool &isSupported) const{
194  double f = RuleLocal::evalSupport<eff_rule>(order, point[0], x[0], isSupported);
195  if (!isSupported) return 0.0;
196  for(int j=1; j<num_dimensions; j++){
197  f *= RuleLocal::evalSupport<eff_rule>(order, point[j], x[j], isSupported);
198  if (!isSupported) return 0.0;
199  }
200  return f;
201  }
202  template<RuleLocal::erule effrule>
203  void diffBasisSupported(const int point[], const double x[], double diff_values[], bool &isSupported) const{
204  isSupported = false;
205  for(int i=0; i<num_dimensions; i++) diff_values[i] = 1.0;
206  bool isDimSupported = false;
207  for(int k=0; k<num_dimensions; k++) {
208  double fval = RuleLocal::evalSupport<effrule>(order, point[k], x[k], isDimSupported);
209  isSupported = isDimSupported or isSupported;
210  for(int j=0; j<k; j++) diff_values[j] *= fval;
211  for(int j=k+1; j<num_dimensions; j++) diff_values[j] *= fval;
212  }
213  for (int k=0; k<num_dimensions; k++) {
214  diff_values[k] *= RuleLocal::diffSupport<effrule>(order, point[k], x[k], isDimSupported);
215  isSupported = isDimSupported or isSupported;
216  }
217  }

234  template<int mode, RuleLocal::erule effrule>
235  void walkTree(const MultiIndexSet &work, const double x[], std::vector<int> &sindx, std::vector<double> &svals, double *y) const{
236  std::vector<int> monkey_count(top_level+1); // traverse the tree, counts the branches of the current node
237  std::vector<int> monkey_tail(top_level+1); // traverse the tree, keeps track of the previous node (history)
238 
239  bool isSupported;
240  double basis_value;
241  std::vector<double> basis_derivative(num_dimensions);
242 
243  for(const auto &r : roots){
244  if (mode == 3 or mode == 4) {
245  diffBasisSupported<effrule>(work.getIndex(r), x, basis_derivative.data(), isSupported);
246  } else {
247  basis_value = evalBasisSupported<effrule>(work.getIndex(r), x, isSupported);
248  }
249 
250  if (isSupported){
251  if (mode == 0){
252  double const *s = surpluses.getStrip(r);
253  for(int k=0; k<num_outputs; k++)
254  y[k] += basis_value * s[k];
255  }else if (mode == 1 or mode == 2){
256  sindx.push_back(r);
257  svals.push_back(basis_value);
258  }else if (mode == 3){
259  double const *s = surpluses.getStrip(r);
260  for(int k=0; k<num_outputs; k++)
261  for (int d=0; d<num_dimensions; d++)
262  y[k * num_dimensions + d] += basis_derivative[d] * s[k];
263  }else{
264  sindx.push_back(r);
265  for (auto dx : basis_derivative)
266  svals.push_back(dx);
267  }
268 
269  int current = 0;
270  monkey_tail[0] = r;
271  monkey_count[0] = pntr[r];
272 
273  while(monkey_count[0] < pntr[monkey_tail[0]+1]){
274  if (monkey_count[current] < pntr[monkey_tail[current]+1]){
275  int p = indx[monkey_count[current]];
276  if (mode == 3 or mode == 4){
277  diffBasisSupported<effrule>(work.getIndex(p), x, basis_derivative.data(), isSupported);
278  }else{
279  basis_value = evalBasisSupported<effrule>(work.getIndex(p), x, isSupported);
280  }
281  if (isSupported){
282  if (mode == 0){
283  double const *s = surpluses.getStrip(p);
284  for(int k=0; k<num_outputs; k++)
285  y[k] += basis_value * s[k];
286  }else if (mode == 1 or mode == 2){
287  sindx.push_back(p);
288  svals.push_back(basis_value);
289  }else if (mode == 3){
290  double const *s = surpluses.getStrip(p);
291  for(int k=0; k<num_outputs; k++)
292  for (int d=0; d<num_dimensions; d++)
293  y[k * num_dimensions + d] += basis_derivative[d] * s[k];
294  }else{
295  sindx.push_back(p);
296  for (auto dx : basis_derivative)
297  svals.push_back(dx);
298  }
299  monkey_tail[++current] = p;
300  monkey_count[current] = pntr[p];
301  }else{
302  monkey_count[current]++;
303  }
304  }else{
305  monkey_count[--current]++;
306  }
307  }
308  }
309  }

311  // according to https://docs.nvidia.com/cuda/cusparse/index.html#sparse-format
312  // "... it is assumed that the indices are provided in increasing order and that each index appears only once."
313  // This may not be a requirement for cusparseDgemvi(), but it may be that I have not tested it sufficiently
314  // Also, see AccelerationDataGPUFull::cusparseMatveci() for inaccuracies in Nvidia documentation
315  if (mode == 2){
316  std::vector<int> map(sindx);
317  std::iota(map.begin(), map.end(), 0);
318  std::sort(map.begin(), map.end(), [&](int a, int b)->bool{ return (sindx[a] < sindx[b]); });
319 
320  std::vector<int> idx = sindx;
321  std::vector<double> vls = svals;
322  std::transform(map.begin(), map.end(), sindx.begin(), [&](int i)->int{ return idx[i]; });
323  std::transform(map.begin(), map.end(), svals.begin(), [&](int i)->double{ return vls[i]; });
324  }
325  }

329  switch(effective_rule) {
330  case RuleLocal::erule::pwc:

333  case RuleLocal::erule::localp:

388  // synchronize with tasgpu_devalpwpoly_feval
389  template<int ord, TypeOneDRule crule, typename T>
390  Data2D<T> encodeSupportForGPU(const MultiIndexSet &work) const{
391  Data2D<T> cpu_support(num_dimensions, work.getNumIndexes());
392  for(int i=0; i<work.getNumIndexes(); i++){
393  const int* p = work.getIndex(i);
394  T *s = cpu_support.getStrip(i);
395  for(int j=0; j<num_dimensions; j++){
396  if (ord == 0){
397  s[j] = static_cast<T>(RuleLocal::getSupport<RuleLocal::erule::pwc>(p[j]));
398  } else {
399  switch(crule) {
400  case rule_localp:
401  s[j] = static_cast<T>(RuleLocal::getSupport<RuleLocal::erule::localp>(p[j]));
402  break;
403  case rule_semilocalp:
404  s[j] = static_cast<T>(RuleLocal::getSupport<RuleLocal::erule::semilocalp>(p[j]));
405  break;
406  case rule_localp0:
407  s[j] = static_cast<T>(RuleLocal::getSupport<RuleLocal::erule::localp0>(p[j]));
408  break;
409  case rule_localpb:
410  s[j] = static_cast<T>(RuleLocal::getSupport<RuleLocal::erule::localpb>(p[j]));
411  break;
412  };
413  if (ord == 2) s[j] *= s[j];
414  if ((crule == rule_localp) || (crule == rule_semilocalp)) if (p[j] == 0) s[j] = static_cast<T>(-1.0); // constant function
415  if ((crule == rule_localp) && (ord == 2)){
416  if (p[j] == 1) s[j] = static_cast<T>(-2.0);
417  else if (p[j] == 2) s[j] = static_cast<T>(-3.0);
418  }
419  if ((crule == rule_semilocalp) && (ord == 2)){
420  if (p[j] == 1) s[j] = static_cast<T>(-4.0);
421  else if (p[j] == 2) s[j] = static_cast<T>(-5.0);
422  }
423  if ((crule == rule_localpb) && (ord == 2)){
424  if (p[j] < 2) s[j] = static_cast<T>(-2.0); // linear functions on level 0
425  }
426  }
427  }
428  }
429  return cpu_support;
430  }

460  }else{

462  if (IO::readFlag<iomode>(is))

464  }

488  }
489  }
490 
491  if (grid->num_outputs > 0) grid->values = StorageSet(is, iomode());
492 
493  return grid;
494  }
