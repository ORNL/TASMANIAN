742  ChangeType favorSparse(bool favor){
743  AlgorithmPreference new_preference = [=]()->AlgorithmPreference{
744  if (favor){
745  return (algorithm_select == algorithm_dense) ? algorithm_autoselect : algorithm_sparse;
746  }else{
747  return (algorithm_select == algorithm_sparse) ? algorithm_autoselect : algorithm_dense;
748  }
749  }();
750  if (new_preference != algorithm_select){
751  algorithm_select = new_preference;
752  return change_sparse_dense;
753  }else{
754  return change_none;
755  }

777  ChangeType enable(TypeAcceleration acc, int new_gpu_id){
778  // get the effective new acceleration mode (use the fallback if acc is not enabled)
779  TypeAcceleration effective_acc = AccelerationMeta::getAvailableFallback(acc);
780  // if switching to a GPU mode, check if the device id is valid
781  if (AccelerationMeta::isAccTypeGPU(effective_acc) and ((new_gpu_id < 0 or new_gpu_id >= AccelerationMeta::getNumGpuDevices())))
782  throw std::runtime_error("Invalid GPU device ID, see ./tasgrid -v for list of detected devices.");
783 
784  // assign the new values for the mode and device, but remember the current gpu state and check whether something changed
785  bool was_on_gpu = on_gpu();
786  ChangeType mode_change = (effective_acc == Utils::exchange(mode, effective_acc)) ? change_none : change_cpu_blas;
787  ChangeType device_change = (new_gpu_id == Utils::exchange(device, new_gpu_id)) ? change_none : change_gpu_device;
788 
789  if (AccelerationMeta::isAccTypeGPU(mode)){
790  // if the new mode is GPU-based, reset the engine and the handles (if already created)
791  engine = Utils::make_unique<GpuEngine>();
792  }else{
793  engine.reset();
794  }
795 
796  if (was_on_gpu){
797  // switching from gpu to gpu issues a change if the device is different
798  // switching from gpu to non-gpu acts as if the device has changed
799  return (on_gpu()) ? device_change : change_gpu_device;
800  }else{
801  // switching from non-gpu to gpu triggers gpu-enabled switch
802  // switching from non-gpu to non-gpu triggers blas change if the modes were different
803  return (on_gpu()) ? change_gpu_enabled : mode_change;
804  }
805  }
807  void setDevice() const{ AccelerationMeta::setDefaultGpuDevice(device); }
809  operator GpuEngine* () const{ return engine.get(); }
811  bool on_gpu() const{ return !!engine; }
